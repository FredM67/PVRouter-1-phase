164void initializeDisplayHW()
165{
166 // Configure the IO drivers for the 4-digit display
167 pinMode(decimalPointLine, OUTPUT); // The 'decimal point' line
168
169 // Set up the control lines for the 74HC4543 7-segment display driver
170 for (int16_t i = 0; i < noOfDigitSelectionLines; ++i)
171 {
172 pinMode(digitSelectionLine[i], OUTPUT);
173 }
174
175 // Set up the enable line for the 74HC4543 7-segment display driver
176 pinMode(enableDisableLine, OUTPUT);
177 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
178
179 // Set up the control lines for the 74HC138 2->4 demultiplexer
180 for (int16_t i = 0; i < noOfDigitLocationLines; ++i)
181 {
182 pinMode(digitLocationLine[i], OUTPUT);
183 }
184}

202void initializeDisplaySW()

204 // Set the pin mode for each segment drive pin

269void configureValueForDisplay(const bool _EDD_isActive, const uint16_t _ValueToDisplay)
270{
271 if constexpr (!(TYPE_OF_DISPLAY == DisplayType::SEG || TYPE_OF_DISPLAY == DisplayType::SEG_HW))
272 {
273 return;
274 }
275
276 static uint8_t locationOfDot{ 0 };
277
278 if (!_EDD_isActive)
279 {
280 // "walking dots" display
281 charsForDisplay[locationOfDot] = 20; // blank
282
283 if (++locationOfDot == noOfDigitLocations)
284 {
285 locationOfDot = 0;
286 }
287
288 charsForDisplay[locationOfDot] = 21; // dot
289
290 return;
291 }
292
293 const auto energyValueExceeds10kWh{ _ValueToDisplay > 9999U };
294
295 uint32_t tmpVal{ energyValueExceeds10kWh ? divu10(_ValueToDisplay) : _ValueToDisplay };
296
297 divmod10(tmpVal, tmpVal, charsForDisplay[3]);
298
299 divmod10(tmpVal, tmpVal, charsForDisplay[2]);
300
301 divmod10(tmpVal, tmpVal, charsForDisplay[1]);
302
303 // Extract the thousands place
304 charsForDisplay[0] = tmpVal;
305
306 // assign the decimal point location
307 if (energyValueExceeds10kWh)
308 {
309 charsForDisplay[1] += 10;
310 } // dec point after 2nd digit
311 else
312 {
313 charsForDisplay[0] += 10;
314 } // dec point after 1st digit
315}

330void update7SegmentHWDisplay()
331{
332 static uint8_t digitLocationThatIsActive = 0;
333
334 // 1. Disable the Decimal Point driver line
335 setPinState(decimalPointLine, LOW);
336
337 // 2. Disable the driver chip while changes are taking place
338 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
339
340 // 3. Determine the next digit location to be active
341 if (++digitLocationThatIsActive == noOfDigitLocations)
342 {
343 digitLocationThatIsActive = 0;
344 }
345
346 // 4. Set up the digit location drivers for the new active location
347 for (uint8_t line = 0; line < noOfDigitLocationLines; ++line)
348 {
349 const auto lineState{ digitLocationMap[digitLocationThatIsActive][line] };
350 setPinState(digitLocationLine[line], lineState);
351 }
352
353 // 5. Determine the character to be displayed at this new location
354 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
355
356 // 6. Configure the 7-segment driver for the character to be displayed
357 for (uint8_t line = 0; line < noOfDigitSelectionLines; ++line)
358 {
359 const auto lineState{ digitValueMap[digitVal][line] };
360 setPinState(digitSelectionLine[line], lineState);
361 }
362
363 // 7. Set up the Decimal Point driver line
364 setPinState(decimalPointLine, digitValueMap[digitVal][DPstatus_columnID]);
365
366 // 8. Enable the 7-segment driver chip
367 setPinState(enableDisableLine, DRIVER_CHIP_ENABLED);
368}

388void update7SegmentSWDisplay()
389{
390 static uint8_t digitLocationThatIsActive{ 0 };
391
392 // 1. Deactivate the location which is currently being displayed
393 setPinState(digitSelectorPin[digitLocationThatIsActive], (uint8_t)DigitEnableStates::DIGIT_DISABLED);
394
395 // 2. Determine the next digit location to be displayed
396 if (++digitLocationThatIsActive == noOfDigitLocations)
397 {
398 digitLocationThatIsActive = 0;
399 }
400
401 // 3. Determine the relevant character for the new active location
402 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
403
404 // 4. Set up the segment drivers for the character to be displayed (includes the DP)
405 for (uint8_t segment = 0; segment < noOfSegmentsPerDigit; ++segment)
406 {
407 const auto segmentState{ segMap[digitVal][segment] };
408 setPinState(segmentDrivePin[segment], segmentState);
409 }
410
411 // 5. Activate the digit-enable line for the new active location
412 setPinState(digitSelectorPin[digitLocationThatIsActive], (uint8_t)DigitEnableStates::DIGIT_ENABLED);
413}
