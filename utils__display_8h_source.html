130void initializeDisplay()
131{
132 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG_HW)
133 {
134 // configure the IO drivers for the 4-digit display
135 //
136 // the Decimal Point line is driven directly from the processor
137 pinMode(decimalPointLine, OUTPUT); // the 'decimal point' line
138
139 // set up the control lines for the 74HC4543 7-seg display driver
140 for (int16_t i = 0; i < noOfDigitSelectionLines; ++i)
141 {
142 pinMode(digitSelectionLine[i], OUTPUT);
143 }
144
145 // an enable line is required for the 74HC4543 7-seg display driver
146 pinMode(enableDisableLine, OUTPUT); // for the 74HC4543 7-seg display driver
147 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
148
149 // set up the control lines for the 74HC138 2->4 demux
150 for (int16_t i = 0; i < noOfDigitLocationLines; ++i)
151 {
152 pinMode(digitLocationLine[i], OUTPUT);
153 }
154 }
155 else if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG)
156 {
157 for (int16_t i = 0; i < noOfSegmentsPerDigit; ++i)
158 {
159 pinMode(segmentDrivePin[i], OUTPUT);
160 }
161
162 for (int16_t i = 0; i < noOfDigitLocations; ++i)
163 {
164 pinMode(digitSelectorPin[i], OUTPUT);
165 }
166
167 for (int16_t i = 0; i < noOfDigitLocations; ++i)
168 {
169 setPinState(digitSelectorPin[i], DIGIT_DISABLED);
170 }
171
172 for (int16_t i = 0; i < noOfSegmentsPerDigit; ++i)
173 {
174 setPinState(segmentDrivePin[i], OFF);
175 }
176 }
177}

180void configureValueForDisplay(const bool _EDD_isActive, const uint16_t _)
181{
182 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG || TYPE_OF_DISPLAY == DisplayType::SEG_HW)

184 static uint8_t locationOfDot = 0;
185
186 if (_EDD_isActive)
187 {
188 uint16_t val = _;
189 bool energyValueExceeds10kWh;
190
191 if (val < 10000)
192 {
193 // no need to re-scale (display to 3 DPs)
194 energyValueExceeds10kWh = false;
195 }
196 else
197 {
198 // re-scale is needed (display to 2 DPs)
199 energyValueExceeds10kWh = true;
200 val = divu10(val);
201 }
202
203 uint8_t thisDigit = divu10(divu10(divu10(val)));
204 charsForDisplay[0] = thisDigit;
205 val -= 1000 * thisDigit;
206
207 thisDigit = divu10(divu10(val));
208 charsForDisplay[1] = thisDigit;
209 val -= 100 * thisDigit;
210
211 thisDigit = divu10(val);
212 charsForDisplay[2] = thisDigit;
213 val -= 10 * thisDigit;
214
215 charsForDisplay[3] = val;
216
217 // assign the decimal point location
218 if (energyValueExceeds10kWh)
219 {
220 charsForDisplay[1] += 10;
221 } // dec point after 2nd digit
222 else
223 {
224 charsForDisplay[0] += 10;
225 } // dec point after 1st digit
226 }
227 else
228 {
229 // "walking dots" display
230 charsForDisplay[locationOfDot] = 20; // blank
231
232 ++locationOfDot;
233 if (locationOfDot >= noOfDigitLocations)
234 {
235 locationOfDot = 0;
236 }
237
238 charsForDisplay[locationOfDot] = 21; // dot
239 }

243void refreshDisplay()
244{
245 // This routine keeps track of which digit is being displayed and checks when its
246 // display time has expired. It then makes the necessary adjustments for displaying
247 // the next digit.
248 // The two versions of the hardware require different logic.
249
250 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG_HW)
251 {
252 // With this version of the hardware, care must be taken that all transitory states
253 // are masked out. Note that the enableDisableLine only masks the seven primary
254 // segments, not the Decimal Point line which must therefore be treated separately.
255 // The sequence is:
256 //
257 // 1. set the decimal point line to 'off'
258 // 2. disable the 7-segment driver chip
259 // 3. determine the next location which is to be active
260 // 4. set up the location lines for the new active location
261 // 5. determine the relevant character for the new active location
262 // 6. configure the driver chip for the new character to be displayed
263 // 7. set up decimal point line for the new active location
264 // 8. enable the 7-segment driver chip
265
266 static uint8_t displayTime_count = 0;
267 static uint8_t digitLocationThatIsActive = 0;
268
269 ++displayTime_count;
270
271 if (displayTime_count > MAX_DISPLAY_TIME_COUNT)
272 {
273 displayTime_count = 0;
274
275 // 1. disable the Decimal Point driver line;
276 setPinState(decimalPointLine, LOW);
277
278 // 2. disable the driver chip while changes are taking place
279 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
280
281 // 3. determine the next digit location to be active
282 ++digitLocationThatIsActive;
283 if (digitLocationThatIsActive >= noOfDigitLocations)
284 {
285 digitLocationThatIsActive = 0;
286 }
287
288 // 4. set up the digit location drivers for the new active location
289 for (uint8_t line = 0; line < noOfDigitLocationLines; ++line)
290 {
291 const auto lineState{ digitLocationMap[digitLocationThatIsActive][line] };
292 setPinState(digitLocationLine[line], lineState);
293 }
294
295 // 5. determine the character to be displayed at this new location
296 // (which includes the decimal point information)
297 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
298
299 // 6. configure the 7-segment driver for the character to be displayed
300 for (uint8_t line = 0; line < noOfDigitSelectionLines; ++line)
301 {
302 const auto lineState{ digitValueMap[digitVal][line] };
303 setPinState(digitSelectionLine[line], lineState);
304 }
305
306 // 7. set up the Decimal Point driver line;
307 setPinState(decimalPointLine, digitValueMap[digitVal][DPstatus_columnID]);
308
309 // 8. enable the 7-segment driver chip
310 setPinState(enableDisableLine, DRIVER_CHIP_ENABLED);
311 }
312 }
313 else if (TYPE_OF_DISPLAY == DisplayType::SEG)
314 {
315 // This version is more straightforward because the digit-enable lines can be
316 // used to mask out all of the transitory states, including the Decimal Point.
317 // The sequence is:
318 //
319 // 1. de-activate the digit-enable line that was previously active
320 // 2. determine the next location which is to be active
321 // 3. determine the relevant character for the new active location
322 // 4. set up the segment drivers for the character to be displayed (includes the DP)
323 // 5. activate the digit-enable line for the new active location
324
325 static uint8_t displayTime_count{ 0 };
326 static uint8_t digitLocationThatIsActive{ 0 };
327
328 ++displayTime_count;
329
330 if (displayTime_count > MAX_DISPLAY_TIME_COUNT)
331 {
332 displayTime_count = 0;
333
334 // 1. de-activate the location which is currently being displayed
335 setPinState(digitSelectorPin[digitLocationThatIsActive], DIGIT_DISABLED);
336
337 // 2. determine the next digit location which is to be displayed
338 ++digitLocationThatIsActive;
339 if (digitLocationThatIsActive >= noOfDigitLocations)
340 {
341 digitLocationThatIsActive = 0;
342 }
343
344 // 3. determine the relevant character for the new active location
345 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
346
347 // 4. set up the segment drivers for the character to be displayed (includes the DP)
348 for (uint8_t segment = 0; segment < noOfSegmentsPerDigit; ++segment)
349 {
350 uint8_t segmentState = segMap[digitVal][segment];
351 setPinState(segmentDrivePin[segment], segmentState);
352 }
353
354 // 5. activate the digit-enable line for the new active location
355 setPinState(digitSelectorPin[digitLocationThatIsActive], DIGIT_ENABLED);
356 }
357 }
358} // end of refreshDisplay()
