158void initializeDisplay()
159{
160 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG_HW)
161 {
162 // configure the IO drivers for the 4-digit display
163 //
164 // the Decimal Point line is driven directly from the processor
165 pinMode(decimalPointLine, OUTPUT); // the 'decimal point' line
166
167 // set up the control lines for the 74HC4543 7-seg display driver
168 for (int16_t i = 0; i < noOfDigitSelectionLines; ++i)
169 {
170 pinMode(digitSelectionLine[i], OUTPUT);
171 }
172
173 // an enable line is required for the 74HC4543 7-seg display driver
174 pinMode(enableDisableLine, OUTPUT); // for the 74HC4543 7-seg display driver
175 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
176
177 // set up the control lines for the 74HC138 2->4 demux
178 for (int16_t i = 0; i < noOfDigitLocationLines; ++i)
179 {
180 pinMode(digitLocationLine[i], OUTPUT);
181 }
182 }
183 else if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG)
184 {
185 for (int16_t i = 0; i < noOfSegmentsPerDigit; ++i)
186 {
187 pinMode(segmentDrivePin[i], OUTPUT);
188 }
189
190 for (int16_t i = 0; i < noOfDigitLocations; ++i)
191 {
192 pinMode(digitSelectorPin[i], OUTPUT);
193 }
194
195 for (int16_t i = 0; i < noOfDigitLocations; ++i)
196 {
197 setPinState(digitSelectorPin[i], DIGIT_DISABLED);
198 }
199
200 for (int16_t i = 0; i < noOfSegmentsPerDigit; ++i)
201 {
202 setPinState(segmentDrivePin[i], OFF);
203 }
204 }
205}

227void configureValueForDisplay(const bool _EDD_isActive, const uint16_t _ValueToDisplay)
228{
229 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG || TYPE_OF_DISPLAY == DisplayType::SEG_HW)
230 {
231 static uint8_t locationOfDot = 0;
232
233 if (_EDD_isActive)
234 {
235 uint16_t val = _ValueToDisplay;
236 bool energyValueExceeds10kWh;
237
238 if (val < 10000)
239 {
240 // no need to re-scale (display to 3 DPs)
241 energyValueExceeds10kWh = false;
242 }
243 else
244 {
245 // re-scale is needed (display to 2 DPs)
246 energyValueExceeds10kWh = true;
247 val = divu10(val);
248 }
249
250 uint8_t thisDigit = divu10(divu10(divu10(val)));
251 charsForDisplay[0] = thisDigit;
252 val -= 1000 * thisDigit;
253
254 thisDigit = divu10(divu10(val));
255 charsForDisplay[1] = thisDigit;
256 val -= 100 * thisDigit;
257
258 thisDigit = divu10(val);
259 charsForDisplay[2] = thisDigit;
260 val -= 10 * thisDigit;
261
262 charsForDisplay[3] = val;
263
264 // assign the decimal point location
265 if (energyValueExceeds10kWh)
266 {
267 charsForDisplay[1] += 10;
268 } // dec point after 2nd digit
269 else
270 {
271 charsForDisplay[0] += 10;
272 } // dec point after 1st digit
273 }
274 else
275 {
276 // "walking dots" display
277 charsForDisplay[locationOfDot] = 20; // blank
278
279 ++locationOfDot;
280 if (locationOfDot >= noOfDigitLocations)
281 {
282 locationOfDot = 0;
283 }
284
285 charsForDisplay[locationOfDot] = 21; // dot
286 }
287 }

321void refreshDisplay()
322{
323 // This routine keeps track of which digit is being displayed and checks when its
324 // display time has expired. It then makes the necessary adjustments for displaying
325 // the next digit.
326 // The two versions of the hardware require different logic.
327
328 if constexpr (TYPE_OF_DISPLAY == DisplayType::SEG_HW)
329 {
330 // With this version of the hardware, care must be taken that all transitory states
331 // are masked out. Note that the enableDisableLine only masks the seven primary
332 // segments, not the Decimal Point line which must therefore be treated separately.
333 // The sequence is:
334 //
335 // 1. set the decimal point line to 'off'
336 // 2. disable the 7-segment driver chip
337 // 3. determine the next location which is to be active
338 // 4. set up the location lines for the new active location
339 // 5. determine the relevant character for the new active location
340 // 6. configure the driver chip for the new character to be displayed
341 // 7. set up decimal point line for the new active location
342 // 8. enable the 7-segment driver chip
343
344 static uint8_t displayTime_count = 0;
345 static uint8_t digitLocationThatIsActive = 0;
346
347 ++displayTime_count;
348
349 if (displayTime_count > MAX_DISPLAY_TIME_COUNT)
350 {
351 displayTime_count = 0;
352
353 // 1. disable the Decimal Point driver line;
354 setPinState(decimalPointLine, LOW);
355
356 // 2. disable the driver chip while changes are taking place
357 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
358
359 // 3. determine the next digit location to be active
360 ++digitLocationThatIsActive;
361 if (digitLocationThatIsActive >= noOfDigitLocations)
362 {
363 digitLocationThatIsActive = 0;
364 }
365
366 // 4. set up the digit location drivers for the new active location
367 for (uint8_t line = 0; line < noOfDigitLocationLines; ++line)
368 {
369 const auto lineState{ digitLocationMap[digitLocationThatIsActive][line] };
370 setPinState(digitLocationLine[line], lineState);
371 }
372
373 // 5. determine the character to be displayed at this new location
374 // (which includes the decimal point information)
375 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
376
377 // 6. configure the 7-segment driver for the character to be displayed
378 for (uint8_t line = 0; line < noOfDigitSelectionLines; ++line)
379 {
380 const auto lineState{ digitValueMap[digitVal][line] };
381 setPinState(digitSelectionLine[line], lineState);
382 }
383
384 // 7. set up the Decimal Point driver line;
385 setPinState(decimalPointLine, digitValueMap[digitVal][DPstatus_columnID]);
386
387 // 8. enable the 7-segment driver chip
388 setPinState(enableDisableLine, DRIVER_CHIP_ENABLED);
389 }
390 }
391 else if (TYPE_OF_DISPLAY == DisplayType::SEG)
392 {
393 // This version is more straightforward because the digit-enable lines can be
394 // used to mask out all of the transitory states, including the Decimal Point.
395 // The sequence is:
396 //
397 // 1. de-activate the digit-enable line that was previously active
398 // 2. determine the next location which is to be active
399 // 3. determine the relevant character for the new active location
400 // 4. set up the segment drivers for the character to be displayed (includes the DP)
401 // 5. activate the digit-enable line for the new active location
402
403 static uint8_t displayTime_count{ 0 };
404 static uint8_t digitLocationThatIsActive{ 0 };
405
406 ++displayTime_count;
407
408 if (displayTime_count > MAX_DISPLAY_TIME_COUNT)
409 {
410 displayTime_count = 0;
411
412 // 1. de-activate the location which is currently being displayed
413 setPinState(digitSelectorPin[digitLocationThatIsActive], DIGIT_DISABLED);
414
415 // 2. determine the next digit location which is to be displayed
416 ++digitLocationThatIsActive;
417 if (digitLocationThatIsActive >= noOfDigitLocations)
418 {
419 digitLocationThatIsActive = 0;
420 }
421
422 // 3. determine the relevant character for the new active location
423 const auto digitVal{ charsForDisplay[digitLocationThatIsActive] };
424
425 // 4. set up the segment drivers for the character to be displayed (includes the DP)
426 for (uint8_t segment = 0; segment < noOfSegmentsPerDigit; ++segment)
427 {
428 uint8_t segmentState = segMap[digitVal][segment];
429 setPinState(segmentDrivePin[segment], segmentState);
430 }
431
432 // 5. activate the digit-enable line for the new active location
433 setPinState(digitSelectorPin[digitLocationThatIsActive], DIGIT_ENABLED);
434 }
435 }
436} // end of refreshDisplay()
