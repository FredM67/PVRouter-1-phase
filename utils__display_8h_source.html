103inline void initializeDisplayHW()
104{
105 // Configure the IO drivers for the 4-digit display
106 pinMode(decimalPointLine, OUTPUT); // The 'decimal point' line
107
108 // Set up the control lines for the 74HC4543 7-segment display driver
109 for (uint8_t i = 0; i < noOfDigitSelectionLines; ++i)
110 {
111 pinMode(digitSelectionLine[i], OUTPUT);
112 }
113
114 // Set up the enable line for the 74HC4543 7-segment display driver
115 pinMode(enableDisableLine, OUTPUT);
116 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
117
118 // Set up the control lines for the 74HC138 2->4 demultiplexer
119 for (uint8_t i = 0; i < noOfDigitLocationLines; ++i)
120 {
121 pinMode(digitLocationLine[i], OUTPUT);
122 }
123}

138inline void update7SegmentHWDisplay()
139{
140 static uint8_t digitLocationThatIsActive = 0;
141
142 // 1. Disable the Decimal Point driver line
143 setPinState(decimalPointLine, LOW);
144
145 // 2. Disable the driver chip while changes are taking place
146 setPinState(enableDisableLine, DRIVER_CHIP_DISABLED);
147
148 // 3. Determine the next digit location to be active
149 if (++digitLocationThatIsActive == noOfDigitLocations)
150 {
151 digitLocationThatIsActive = 0;
152 }
153
154 // 4. Set up the digit location drivers for the new active location
155 for (uint8_t line = 0; line < noOfDigitLocationLines; ++line)
156 {
157 const auto& lineState{ digitLocationMap[digitLocationThatIsActive][line] };
158 setPinState(digitLocationLine[line], lineState);
159 }
160
161 // 5. Determine the character to be displayed at this new location
162 const auto& digitVal{ charsForDisplay[digitLocationThatIsActive] };
163
164 // 6. Configure the 7-segment driver for the character to be displayed
165 for (uint8_t line = 0; line < noOfDigitSelectionLines; ++line)
166 {
167 const auto& lineState{ digitValueMap[digitVal & 0x7F][line] };
168 setPinState(digitSelectionLine[line], lineState);
169 }
170
171 // 7. Set up the Decimal Point driver line
172 setPinState(decimalPointLine, digitVal & 0x80); // DP
173
174
175 // 8. Enable the 7-segment driver chip
176 setPinState(enableDisableLine, DRIVER_CHIP_ENABLED);
177}

234inline void initializeDisplaySW()
235{
236 // Set the pin mode for each segment drive pin
237 for (uint8_t i = 0; i < noOfSegmentsPerDigit; ++i)
238 {
239 pinMode(segmentDrivePin[i], OUTPUT);
240 }
241
242 // Set the pin mode for each digit selector pin
243 for (uint8_t i = 0; i < noOfDigitLocations; ++i)
244 {
245 pinMode(digitSelectorPin[i], OUTPUT);
246 }
247
248 // Disable all digit selector pins initially
249 for (uint8_t i = 0; i < noOfDigitLocations; ++i)
250 {
251 setPinState(digitSelectorPin[i], (uint8_t)DigitEnableStates::DIGIT_DISABLED);
252 }
253
254 // Turn off all segment drive pins initially
255 for (uint8_t i = 0; i < noOfSegmentsPerDigit; ++i)
256 {
257 setPinState(segmentDrivePin[i], OFF);
258 }

279inline void update7SegmentSWDisplay()
280{
281 static uint8_t digitLocationThatIsActive{ 0 };
282
283 // 1. Deactivate the location which is currently being displayed
284 setPinState(digitSelectorPin[digitLocationThatIsActive], (uint8_t)DigitEnableStates::DIGIT_DISABLED);
285
286 // 2. Determine the next digit location to be displayed
287 if (++digitLocationThatIsActive == noOfDigitLocations)
288 {
289 digitLocationThatIsActive = 0;
290 }
291
292 // 3. Determine the relevant character for the new active location
293 const auto& digitVal{ charsForDisplay[digitLocationThatIsActive] };
294
295 // 4. Set up the segment drivers for the character to be displayed (includes the DP)
296 for (uint8_t segment = 0; segment < noOfSegmentsPerDigit - 1; ++segment)
297 {
298 const auto& segmentState{ segMap[digitVal & 0x7F][segment] };
299 setPinState(segmentDrivePin[segment], segmentState);
300 }
301 setPinState(segmentDrivePin[noOfSegmentsPerDigit - 1], digitVal & 0x80); // DP
302
303
304 // 5. Activate the digit-enable line for the new active location
305 setPinState(digitSelectorPin[digitLocationThatIsActive], (uint8_t)DigitEnableStates::DIGIT_ENABLED);
306}

355inline void configureValueForDisplay(bool _EDD_isActive, const uint16_t _ValueToDisplay)
356{
357 if constexpr (!(TYPE_OF_DISPLAY == DisplayType::SEG || TYPE_OF_DISPLAY == DisplayType::SEG_HW))
358 {
359 return;
360 }
361
362 if (!_EDD_isActive)
363 {
364 static uint8_t locationOfDot{ 0 };
365
366 // "walking dots" display
367 charsForDisplay[0] = 10;
368 charsForDisplay[1] = 10;
369 charsForDisplay[2] = 10;
370 charsForDisplay[3] = 10;
371
372 if (++locationOfDot == noOfDigitLocations)
373 {
374 locationOfDot = 0;
375 }
376
377 charsForDisplay[locationOfDot] |= 0x80; // dot
378
379 return;
380 }
381
382 const auto energyValueExceeds10kWh{ _ValueToDisplay > 9999U };
383
384 uint32_t tmpVal{ energyValueExceeds10kWh ? divu10(_ValueToDisplay) : _ValueToDisplay };
385
386 divmod10(tmpVal, tmpVal, charsForDisplay[3]); // Extract units place
387 divmod10(tmpVal, tmpVal, charsForDisplay[2]); // Extract tens place
388 divmod10(tmpVal, tmpVal, charsForDisplay[1]); // Extract hundreds place
389 charsForDisplay[0] = tmpVal; // Remaining value is the thousands place
390
391 // assign the decimal point location
392 if (energyValueExceeds10kWh)
393 {
394 charsForDisplay[1] |= 0x80;
395 } // dec point after 2nd digit
396 else
397 {
398 charsForDisplay[0] |= 0x80;
399 } // dec point after 1st digit
400}

422 // This routine keeps track of which digit is being displayed and checks when its
