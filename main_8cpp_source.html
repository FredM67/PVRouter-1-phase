159bool proceedLoadPrioritiesAndOverridingDualTariff(const int16_t currentTemperature_x100)
160{
161 constexpr int16_t iTemperatureThreshold_x100{ iTemperatureThreshold * 100 };
162 static bool pinOffPeakState{ HIGH };
163 const auto pinNewState{ getPinState(dualTariffPin) };
164
165 if (pinOffPeakState && !pinNewState)
166 {
167 // we start off-peak period
168 DBUGLN(F("Change to off-peak period!"));
169
170 ul_TimeOffPeak = millis();
171
172 if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
173 {
174 proceedRotation();
175 }
176 }
177 else
178 {
179 const auto ulElapsedTime{ static_cast< uint32_t >(millis() - ul_TimeOffPeak) };
180 const auto pinState{ getPinState(forcePin) };
181
182 for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
183 {
184 // for each load, if we're inside off-peak period and within the 'force period', trigger the ISR to turn the load ON
185 if (!pinOffPeakState && !pinNewState && (ulElapsedTime >= rg_OffsetForce[i][0]) && (ulElapsedTime < rg_OffsetForce[i][1]))
186 {
187 b_overrideLoadOn[i] = !pinState || (currentTemperature_x100 <= iTemperatureThreshold_x100);
188 }
189 else
190 {
191 b_overrideLoadOn[i] = !pinState;
192 }
193 }
194 }
195 // end of off-peak period
196 if (!pinOffPeakState && pinNewState)
197 {
198 DBUGLN(F("Change to peak period!"));
199 }
200
201 pinOffPeakState = pinNewState;
202
203 return (LOW == pinOffPeakState);
204}

215bool proceedLoadPrioritiesAndOverriding(const int16_t currentTemperature_x100)
216{
217 if constexpr (DUAL_TARIFF)
218 {
219 return proceedLoadPrioritiesAndOverridingDualTariff(currentTemperature_x100);
220 }
221
222 if constexpr (EMONESP_CONTROL)
223 {
224 static uint8_t pinRotationState{ HIGH };
225 const auto pinNewState{ getPinState(rotationPin) };
226
227 if (pinRotationState && !pinNewState)
228 {
229 DBUGLN(F("Trigger rotation!"));
230
231 proceedRotation();
232 }
233 pinRotationState = pinNewState;
234 }
235 else if constexpr (PRIORITY_ROTATION == RotationModes::AUTO)
236 {
237 if (ROTATION_AFTER_CYCLES < absenceOfDivertedEnergyCount)
238 {
239 proceedRotation();
240
241 absenceOfDivertedEnergyCount = 0;
242 }
243 }
244
245 if constexpr (OVERRIDE_PIN_PRESENT)
246 {
247 const auto pinState{ getPinState(forcePin) };
248
249 for (auto &bOverrideLoad : b_overrideLoadOn)
250 {
251 bOverrideLoad = !pinState;
252 }
253 }
254
255 return false;
256}

286void updateTemperature()
287{
288 if constexpr (TEMP_SENSOR_PRESENT)
289 {
290 uint8_t idx{ temperatureSensing.get_size() };
291 do
292 {
293 auto tmp = temperatureSensing.readTemperature(--idx);
294
295 // if read temperature is 85 and the delta with previous is greater than 5, skip the value
296 if (8500 == tmp && (abs(tmp - tx_data.temperature_x100[idx]) > 500))
297 {
298 tmp = DEVICE_DISCONNECTED_RAW;
299 }
300
301 tx_data.temperature_x100[idx] = tmp;
302 } while (idx);
303
304 temperatureSensing.requestTemperatures(); // for use next time around

375{
376 static bool initLoop{ true };

378 static bool bOffPeak{ false };
379 static uint8_t timerForDisplayUpdate{ 0 };
380 static int16_t iTemperature_x100{ 0 };
381
382 if (b_newCycle) // flag is set after every pair of ADC conversions

384 b_newCycle = false; // reset the flag

386 ++timerForDisplayUpdate;
387
388 if (timerForDisplayUpdate >= UPDATE_PERIOD_FOR_DISPLAYED_DATA)
389 { // the 4-digit display needs to be refreshed every few mS. For convenience,
390 // this action is performed every N times around this processing loop.
391 timerForDisplayUpdate = 0;
392
393 // After a pre-defined period of inactivity, the 4-digit display needs to
394 // close down in readiness for the next's day's data.
395 //
396 if (absenceOfDivertedEnergyCount > displayShutdown_inMainsCycles)

398 // clear the accumulators for diverted energy

400 divertedEnergyRecent_IEU = 0;
401 EDD_isActive = false; // energy diversion detector is now inactive
402 }
403
404 configureValueForDisplay(EDD_isActive, divertedEnergyTotal_Wh);
405 // Serial.println(energyInBucket_prediction);
406 }
407
408 if (perSecondTimer >= SUPPLY_FREQUENCY)
409 {
410 perSecondTimer = 0;
411
412 if constexpr (WATCHDOG_PIN_PRESENT)
413 {
414 togglePin(watchDogPin);
415 }
416
417 if (!initLoop)
418 {
419 updateWatchdog();
420 }
421
422 checkDiversionOnOff();
423
424 if (!forceFullPower())
425 {
426 bOffPeak = proceedLoadPrioritiesAndOverriding(iTemperature_x100); // called every second
427 }
428

431 relays.inc_duration();
432 relays.proceed_relays();
433 }

436 }

454 }
455
