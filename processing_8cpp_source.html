212void initializeProcessing()
213{
214 setPinsAsOutput(getOutputPins()); // set the output pins as OUTPUT
215 setPinsAsInputPullup(getInputPins()); // set the input pins as INPUT_PULLUP
216
217 // First stop the ADC
218 bit_clear(ADCSRA, ADEN);
219
220 // Activate free-running mode
221 ADCSRB = 0x00;
222
223 // Set up the ADC to be free-running
224 bit_set(ADCSRA, ADPS0); // Set the ADC's clock to system clock / 128
225 bit_set(ADCSRA, ADPS1);
226 bit_set(ADCSRA, ADPS2);
227
228 bit_set(ADCSRA, ADATE); // set the Auto Trigger Enable bit in the ADCSRA register. Because
229 // bits ADTS0-2 have not been set (i.e. they are all zero), the
230 // ADC's trigger source is set to "free running mode".
231
232 bit_set(ADCSRA, ADIE); // set the ADC interrupt enable bit. When this bit is written
233 // to one and the I-bit in SREG is set, the
234 // ADC Conversion Complete Interrupt is activated.
235
236 bit_set(ADCSRA, ADEN); // Enable the ADC
237
238 bit_set(ADCSRA, ADSC); // start ADC manually first time
239
240 sei(); // Enable Global Interrupts
241}

378void processGridCurrentRawSample(const int16_t rawSample)
379{
380 // extra items for an LPF to improve the processing of data samples from CT1
381 static int32_t lpf_long{}; // new LPF, for offsetting the behaviour of CTx as a HPF
382
383 // First, deal with the power at the grid connection point (as measured via CT1)
384 // remove most of the DC offset from the current sample (the precise value does not matter)
385 int32_t sampleIminusDC_grid = ((int32_t)(rawSample - DCoffset_I)) << 8;
386 //
387 // extra filtering to offset the HPF effect of CT1
388 const int32_t last_lpf_long = lpf_long;
389 lpf_long += alpha * (sampleIminusDC_grid - last_lpf_long);
390 sampleIminusDC_grid += (lpf_gain * lpf_long);
391
392 // calculate the "real power" in this sample pair and add to the accumulated sum
393 const int32_t filtV_div4 = sampleVminusDC_long >> 2; // reduce to 16-bits (now x64, or 2^6)
394 const int32_t filtI_div4 = sampleIminusDC_grid >> 2; // reduce to 16-bits (now x64, or 2^6)
395 int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
396 instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
397 sumP_grid += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
398 sumP_grid_overDL_Period += instP;
399}

492void processRawSamples()
493{
494 if (polarityConfirmed == Polarities::POSITIVE)
495 {
496 if (polarityConfirmedOfLastSampleV != Polarities::POSITIVE)
497 {
498 // This is the start of a new +ve half cycle (just after the zero-crossing point)
499 if (beyondStartUpPeriod)
500 {
501 processPlusHalfCycle();
502
503 processStartNewCycle();
504
505 if (EDD_isActive) // Energy Diversion Display
506 {
507 // For diverted energy, the latest contribution needs to be added to an
508 // accumulator which operates with maximum precision.
509
510 if (realEnergy_diverted < antiCreepLimit_inIEUperMainsCycle)
511 {
512 realEnergy_diverted = 0;
513 }
514 divertedEnergyRecent_IEU += realEnergy_diverted;
515
516 // Whole kWh are then recorded separately
517 if (divertedEnergyRecent_IEU > IEU_per_Wh)
518 {
519 divertedEnergyRecent_IEU -= IEU_per_Wh;
520 if (!b_overrideLoadOn[0])
521 {
522 ++divertedEnergyTotal_Wh;
523 }
524 }
525 }
526 }
527 else
528 {
529 processStartUp();
530 }
531 } // end of processing that is specific to the first Vsample in each +ve half cycle
532
533 // still processing samples where the voltage is Polarities::POSITIVE ...
534 // (in this go-faster code, the action from here has moved to the negative half of the cycle)
535
536 } // end of processing that is specific to samples where the voltage is positive
537 else // the polarity of this sample is negative
538 {
539 if (polarityConfirmedOfLastSampleV != Polarities::NEGATIVE)
540 {
541 // This is the start of a new -ve half cycle (just after the zero-crossing point)
542 processMinusHalfCycle();
543 }
544
545 // check to see whether the trigger device can now be reliably armed
546 if (sampleSetsDuringNegativeHalfOfMainsCycle == 3)
547 {
548 if (beyondStartUpPeriod)
549 {
550 /* Determining whether any of the loads need to be changed is is a 3-stage process:
551 * - change the LOGICAL load states as necessary to maintain the energy level
552 * - update the PHYSICAL load states according to the logical -> physical mapping
553 * - update the driver lines for each of the loads.
554 */
555
556 // Restrictions apply for the period immediately after a load has been switched.
557 // Here the recentTransition flag is checked and updated as necessary.
558 if (recentTransition)
559 {
560 if (++postTransitionCount == POST_TRANSITION_MAX_COUNT)
561 {
562 recentTransition = false;
563 }
564 }
565
566 if (energyInBucket_prediction > midPointOfEnergyBucket_long)
567 {
568 // the energy state is in the upper half of the working range
569 lowerEnergyThreshold = lowerThreshold_default; // reset the "opposite" threshold
570 if (energyInBucket_prediction > upperEnergyThreshold)
571 {
572 // Because the energy level is high, some action may be required
573 proceedHighEnergyLevel();
574 }
575 }
576 else
577 { // the energy state is in the lower half of the working range
578 upperEnergyThreshold = upperThreshold_default; // reset the "opposite" threshold
579 if (energyInBucket_prediction < lowerEnergyThreshold)
580 {
581 // Because the energy level is low, some action may be required
582 proceedLowEnergyLevel();
583 }
584 }
585
586 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
587
588 // update each of the physical loads
589 updatePortsStates();
590
591 // update the Energy Diversion Detector
592 if (loadPrioritiesAndState[0] & loadStateOnBit)
593 {
594 absenceOfDivertedEnergyCount = 0;
595 EDD_isIdle = false;
596 EDD_isActive = true;
597 }
598 else
599 {
600 EDD_isIdle = true;
601 }
602
603 // Now that the energy-related decisions have been taken, min and max limits can now
604 // be applied to the level of the energy bucket. This is to ensure correct operation
605 // when conditions change, i.e. when import changes to export, and vice versa.
606 //
607 if (energyInBucket_long > capacityOfEnergyBucket_long)
608 {
609 energyInBucket_long = capacityOfEnergyBucket_long;
610 }
611 else if (energyInBucket_long < 0)
612 {
613 energyInBucket_long = 0;
614 }
615 }
616 }
617
618 ++sampleSetsDuringNegativeHalfOfMainsCycle;
619 } // end of processing that is specific to samples where the voltage is negative
620 refreshDisplay();
621}

804void processMinusHalfCycle()
805{
806 // This is the start of a new -ve half cycle (just after the zero-crossing point)
807 // which is a convenient point to update the Low Pass Filter for DC-offset removal
808 // The portion which is fed back into the integrator is approximately one percent
809 // of the average offset of all the Vsamples in the previous mains cycle.
810 //
811 long previousOffset = DCoffset_V_long;
812 DCoffset_V_long = previousOffset + (cumVdeltasThisCycle_long >> 12);
813 cumVdeltasThisCycle_long = 0;
814
815 // To ensure that the LPF will always start up correctly when 240V AC is available, its
816 // output value needs to be prevented from drifting beyond the likely range of the
817 // voltage signal. This avoids the need to use a HPF as was done for initial Mk2 builds.
818 //
819 if (DCoffset_V_long < DCoffset_V_min)
820 {
821 DCoffset_V_long = DCoffset_V_min;
822 }
823 else if (DCoffset_V_long > DCoffset_V_max)
824 {
825 DCoffset_V_long = DCoffset_V_max;
826 }
827
828 // The average power that has been measured during the first half of this mains cycle can now be used
829 // to predict the energy state at the end of this mains cycle. That prediction will be used to alter
830 // the state of the load as necessary. The arming signal for the triac can't be set yet - that must
831 // wait until the voltage has advanced further beyond the -ve going zero-crossing point.
832 //
833 long averagePower = sumP_grid / sampleSetsDuringThisMainsCycle; // for 1st half of this mains cycle only
834 //
835 // To avoid repetitive and unnecessary calculations, the increase in energy during each mains cycle is
836 // deemed to be numerically equal to the average power. The predicted value for the energy state at the
837 // end of this mains cycle will therefore be the known energy state at its start plus the average power
838 // as measured. Although the average power has been determined over only half a mains cycle, the correct
839 // number of contributing sample sets has been used so the result can be expected to be a true measurement
840 // of average power, not half of it.
841 //
842 energyInBucket_prediction = energyInBucket_long + averagePower; // at end of this mains cycle
843}

859void processLatestContribution()
860{
861 // Calculate the real power and energy during the last whole mains cycle.
862 //
863 // sumP contains the sum of many individual calculations of instantaneous power. In
864 // order to obtain the average power during the relevant period, sumP must first be
865 // divided by the number of samples that have contributed to its value.
866 //
867 // The next stage would normally be to apply a calibration factor so that real power
868 // can be expressed in Watts. That's fine for floating point maths, but it's not such
869 // a good idea when integer maths is being used. To keep the numbers large, and also
870 // to save time, calibration of power is omitted at this stage. Real Power (stored as
871 // a 'long') is therefore (1/powerCal) times larger than the actual power in Watts.
872 //
873 int32_t realPower_grid = sumP_grid / sampleSetsDuringThisMainsCycle; // proportional to Watts
874 int32_t realPower_diverted = sumP_diverted / sampleSetsDuringThisMainsCycle; // proportional to Watts
875
876 realPower_grid -= requiredExportPerMainsCycle_inIEU; // <- useful for PV simulation
877
878 // Next, the energy content of this power rating needs to be determined. Energy is
879 // power multiplied by time, so the next step would normally be to multiply the measured
880 // value of power by the time over which it was measured.
881 // Instantaneous power is calculated once every mains cycle. When integer maths is
882 // being used, a repetitive power-to-energy conversion seems an unnecessary workload.
883 // As all sampling periods are of similar duration, it is more efficient to just
884 // add all of the power samples together, and note that their sum is actually
885 // SUPPLY_FREQUENCY greater than it would otherwise be.
886 // Although the numerical value itself does not change, I thought that a new name
887 // may be helpful so as to minimise confusion.
888 // The 'energy' variable below is SUPPLY_FREQUENCY * (1/powerCal) times larger than
889 // the actual energy in Joules.
890 //
891 realEnergy_grid = realPower_grid;
892 realEnergy_diverted = realPower_diverted;
893
894 // Energy contributions from the grid connection point (CT1) are summed in an
895 // accumulator which is known as the energy bucket. The purpose of the energy bucket
896 // is to mimic the operation of the supply meter. The range over which energy can
897 // pass to and fro without loss or charge to the user is known as its 'sweet-zone'.
898 // The capacity of the energy bucket is set to this same value within setup().
899 //
900 // The latest contribution can now be added to this energy bucket
901 energyInBucket_long += realEnergy_grid;
902
903 b_newCycle = true; // a 50 Hz 'tick' for use by the main code
904}

925

927 {
928 return;
929 }

935 upperEnergyThreshold = energyInBucket_prediction;

976 const auto tempLoad{ nextLogicalLoadToBeRemoved() };
977
978 if (tempLoad >= NO_OF_DUMPLOADS)
979 {
980 return;
981 }
982
983 // a load which is now ON has been identified for potentially being switched OFF
984 if (recentTransition)
985 {
986 // During the post-transition period, any decrease in the energy level is noted.
987 lowerEnergyThreshold = energyInBucket_prediction;
988
989 // the energy thresholds must remain within range
990 if (lowerEnergyThreshold < 0)
991 {
992 lowerEnergyThreshold = 0;
993 }
994
995 // Only the active load may be switched during this period. All other loads must
996 // wait until the recent transition has had sufficient opportunity to take effect.
997 bOK_toRemoveLoad = (tempLoad == activeLoad);
998 }
999
1000 if (bOK_toRemoveLoad)
1001 {
1002 loadPrioritiesAndState[tempLoad] &= loadStateMask;
1003 activeLoad = tempLoad;
1004 postTransitionCount = 0;
1005 recentTransition = true;
1006 }
1007}
