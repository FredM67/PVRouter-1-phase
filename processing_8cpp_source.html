99{

116 // First stop the ADC
117 bit_clear(ADCSRA, ADEN);
118
119 // Activate free-running mode
120 ADCSRB = 0x00;
121
122 // Set up the ADC to be free-running
123 bit_set(ADCSRA, ADPS0); // Set the ADC's clock to system clock / 128
124 bit_set(ADCSRA, ADPS1);
125 bit_set(ADCSRA, ADPS2);
126

128 // bits ADTS0-2 have not been set (i.e. they are all zero), the
129 // ADC's trigger source is set to "free running mode".

132 // to one and the I-bit in SREG is set, the
133 // ADC Conversion Complete Interrupt is activated.
134
135 bit_set(ADCSRA, ADEN); // Enable the ADC
136
137 bit_set(ADCSRA, ADSC); // start ADC manually first time

301void processGridCurrentRawSample(const int16_t rawSample)
302{
303 // extra items for an LPF to improve the processing of data samples from CT1
304 static int32_t lpf_long{}; // new LPF, for offsetting the behaviour of CTx as a HPF
305
306 // First, deal with the power at the grid connection point (as measured via CT1)
307 // remove most of the DC offset from the current sample (the precise value does not matter)
308 int32_t sampleIminusDC_grid = ((int32_t)(rawSample - DCoffset_I)) << 8;
309 //
310 // extra filtering to offset the HPF effect of CT1
311 const int32_t last_lpf_long = lpf_long;
312 lpf_long += alpha * (sampleIminusDC_grid - last_lpf_long);
313 sampleIminusDC_grid += (lpf_gain * lpf_long);
314
315 // calculate the "real power" in this sample pair and add to the accumulated sum
316 const int32_t filtV_div4 = sampleVminusDC_long >> 2; // reduce to 16-bits (now x64, or 2^6)
317 const int32_t filtI_div4 = sampleIminusDC_grid >> 2; // reduce to 16-bits (now x64, or 2^6)
318 int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
319 instP = instP >> 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
320 sumP_grid += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
321 sumP_grid_overDL_Period += instP;
322}

379void processRawSamples()
380{
381 if (polarityConfirmed == Polarities::POSITIVE)
382 {
383 if (polarityConfirmedOfLastSampleV != Polarities::POSITIVE)
384 {
385 // This is the start of a new +ve half cycle (just after the zero-crossing point)
386 if (beyondStartUpPeriod)
387 {
388 processPlusHalfCycle();
389
390 processStartNewCycle();
391
392 if (EDD_isActive) // Energy Diversion Display
393 {
394 // For diverted energy, the latest contribution needs to be added to an
395 // accumulator which operates with maximum precision.
396
397 if (realEnergy_diverted < antiCreepLimit_inIEUperMainsCycle)
398 {
399 realEnergy_diverted = 0;
400 }
401 divertedEnergyRecent_IEU += realEnergy_diverted;
402
403 // Whole kWh are then recorded separately
404 if (divertedEnergyRecent_IEU > IEU_per_Wh)
405 {
406 divertedEnergyRecent_IEU -= IEU_per_Wh;
407 ++divertedEnergyTotal_Wh;
408 }
409 }
410 }
411 else
412 {
413 processStartUp();
414 }
415 } // end of processing that is specific to the first Vsample in each +ve half cycle
416
417 // still processing samples where the voltage is Polarities::POSITIVE ...
418 // (in this go-faster code, the action from here has moved to the negative half of the cycle)
419
420 } // end of processing that is specific to samples where the voltage is positive
421
422 else // the polarity of this sample is negative
423 {
424 if (polarityConfirmedOfLastSampleV != Polarities::NEGATIVE)
425 {
426 // This is the start of a new -ve half cycle (just after the zero-crossing point)
427 processMinusHalfCycle();
428 }
429
430 // check to see whether the trigger device can now be reliably armed
431 if (sampleSetsDuringNegativeHalfOfMainsCycle == 3)
432 {
433 if (beyondStartUpPeriod)
434 {
435 /* Determining whether any of the loads need to be changed is is a 3-stage process:
436 * - change the LOGICAL load states as necessary to maintain the energy level
437 * - update the PHYSICAL load states according to the logical -> physical mapping
438 * - update the driver lines for each of the loads.
439 */
440
441 // Restrictions apply for the period immediately after a load has been switched.
442 // Here the recentTransition flag is checked and updated as necessary.
443 if (recentTransition)
444 {
445 ++postTransitionCount;
446 if (postTransitionCount >= POST_TRANSITION_MAX_COUNT)
447 {
448 recentTransition = false;
449 }
450 }
451
452 if (energyInBucket_prediction > midPointOfEnergyBucket_long)
453 {
454 // the energy state is in the upper half of the working range
455 lowerEnergyThreshold = lowerThreshold_default; // reset the "opposite" threshold
456 if (energyInBucket_prediction > upperEnergyThreshold)
457 {
458 // Because the energy level is high, some action may be required
459 proceedHighEnergyLevel();
460 }
461 }
462 else
463 { // the energy state is in the lower half of the working range
464 upperEnergyThreshold = upperThreshold_default; // reset the "opposite" threshold
465 if (energyInBucket_prediction < lowerEnergyThreshold)
466 {
467 // Because the energy level is low, some action may be required
468 proceedLowEnergyLevel();
469 }
470 }
471
472 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
473
474 // update each of the physical loads
475 updatePortsStates();
476
477 // update the Energy Diversion Detector
478 if (physicalLoadState[0] == LoadStates::LOAD_ON)
479 {
480 absenceOfDivertedEnergyCount = 0;
481 EDD_isActive = true;
482 }
483 else
484 {
485 ++absenceOfDivertedEnergyCount;
486 }
487
488 // Now that the energy-related decisions have been taken, min and max limits can now
489 // be applied to the level of the energy bucket. This is to ensure correct operation
490 // when conditions change, i.e. when import changes to export, and vice versa.
491 //
492 if (energyInBucket_long > capacityOfEnergyBucket_long)
493 {
494 energyInBucket_long = capacityOfEnergyBucket_long;
495 }
496 else if (energyInBucket_long < 0)
497 {
498 energyInBucket_long = 0;
499 }
500 }
501 }
502
503 ++sampleSetsDuringNegativeHalfOfMainsCycle;
504 } // end of processing that is specific to samples where the voltage is negative
505}

636void processMinusHalfCycle()
637{
638 // This is the start of a new -ve half cycle (just after the zero-crossing point)
639 // which is a convenient point to update the Low Pass Filter for DC-offset removal
640 // The portion which is fed back into the integrator is approximately one percent
641 // of the average offset of all the Vsamples in the previous mains cycle.
642 //
643 long previousOffset = DCoffset_V_long;
644 DCoffset_V_long = previousOffset + (cumVdeltasThisCycle_long >> 12);
645 cumVdeltasThisCycle_long = 0;
646
647 // To ensure that the LPF will always start up correctly when 240V AC is available, its
648 // output value needs to be prevented from drifting beyond the likely range of the
649 // voltage signal. This avoids the need to use a HPF as was done for initial Mk2 builds.
650 //
651 if (DCoffset_V_long < DCoffset_V_min)
652 {
653 DCoffset_V_long = DCoffset_V_min;
654 }
655 else if (DCoffset_V_long > DCoffset_V_max)
656 {
657 DCoffset_V_long = DCoffset_V_max;
658 }
659
660 // The average power that has been measured during the first half of this mains cycle can now be used
661 // to predict the energy state at the end of this mains cycle. That prediction will be used to alter
662 // the state of the load as necessary. The arming signal for the triac can't be set yet - that must
663 // wait until the voltage has advanced further beyond the -ve going zero-crossing point.
664 //
665 long averagePower = sumP_grid / sampleSetsDuringThisMainsCycle; // for 1st half of this mains cycle only
666 //
667 // To avoid repetitive and unnecessary calculations, the increase in energy during each mains cycle is
668 // deemed to be numerically equal to the average power. The predicted value for the energy state at the
669 // end of this mains cycle will therefore be the known energy state at its start plus the average power
670 // as measured. Although the average power has been determined over only half a mains cycle, the correct
671 // number of contributing sample sets has been used so the result can be expected to be a true measurement
672 // of average power, not half of it.
673 //
674 energyInBucket_prediction = energyInBucket_long + averagePower; // at end of this mains cycle
675}

683void processLatestContribution()
684{
685 // Calculate the real power and energy during the last whole mains cycle.
686 //
687 // sumP contains the sum of many individual calculations of instantaneous power. In
688 // order to obtain the average power during the relevant period, sumP must first be
689 // divided by the number of samples that have contributed to its value.
690 //
691 // The next stage would normally be to apply a calibration factor so that real power
692 // can be expressed in Watts. That's fine for floating point maths, but it's not such
693 // a good idea when integer maths is being used. To keep the numbers large, and also
694 // to save time, calibration of power is omitted at this stage. Real Power (stored as
695 // a 'long') is therefore (1/powerCal) times larger than the actual power in Watts.
696 //
697 int32_t realPower_grid = sumP_grid / sampleSetsDuringThisMainsCycle; // proportional to Watts
698 int32_t realPower_diverted = sumP_diverted / sampleSetsDuringThisMainsCycle; // proportional to Watts
699
700 realPower_grid -= requiredExportPerMainsCycle_inIEU; // <- useful for PV simulation
701
702 // Next, the energy content of this power rating needs to be determined. Energy is
703 // power multiplied by time, so the next step would normally be to multiply the measured
704 // value of power by the time over which it was measured.
705 // Instantaneous power is calculated once every mains cycle. When integer maths is
706 // being used, a repetitive power-to-energy conversion seems an unnecessary workload.
707 // As all sampling periods are of similar duration, it is more efficient to just
708 // add all of the power samples together, and note that their sum is actually
709 // SUPPLY_FREQUENCY greater than it would otherwise be.
710 // Although the numerical value itself does not change, I thought that a new name
711 // may be helpful so as to minimise confusion.
712 // The 'energy' variable below is SUPPLY_FREQUENCY * (1/powerCal) times larger than
713 // the actual energy in Joules.
714 //
715 realEnergy_grid = realPower_grid;
716 realEnergy_diverted = realPower_diverted;
717
718 // Energy contributions from the grid connection point (CT1) are summed in an
719 // accumulator which is known as the energy bucket. The purpose of the energy bucket
720 // is to mimic the operation of the supply meter. The range over which energy can
721 // pass to and fro without loss or charge to the user is known as its 'sweet-zone'.
722 // The capacity of the energy bucket is set to this same value within setup().
723 //
724 // The latest contribution can now be added to this energy bucket
725 energyInBucket_long += realEnergy_grid;
726
727 b_newCycle = true; // a 50 Hz 'tick' for use by the main code
728}

735void proceedHighEnergyLevel()
736{
737 bool bOK_toAddLoad{ true };
738 const auto tempLoad{ nextLogicalLoadToBeAdded() };
739
740 if (tempLoad >= NO_OF_DUMPLOADS)
741 {
742 return;
743 }
744
745 // a load which is now OFF has been identified for potentially being switched ON
746 if (recentTransition)
747 {
748 // During the post-transition period, any increase in the energy level is noted.
749 upperEnergyThreshold = energyInBucket_prediction;
750
751 // the energy thresholds must remain within range
752 if (upperEnergyThreshold > capacityOfEnergyBucket_long)
753 {
754 upperEnergyThreshold = capacityOfEnergyBucket_long;
755 }
756
757 // Only the active load may be switched during this period. All other loads must
758 // wait until the recent transition has had sufficient opportunity to take effect.
759 bOK_toAddLoad = (tempLoad == activeLoad);
760 }
761
762 if (bOK_toAddLoad)
763 {
764 loadPrioritiesAndState[tempLoad] |= loadStateOnBit;
765 activeLoad = tempLoad;
766 postTransitionCount = 0;
767 recentTransition = true;
768 }
