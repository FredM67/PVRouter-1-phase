212void initializeProcessing()
213{
214 if constexpr (OLD_PCB)
215 initializeOldPCBPins();
216 else
217 {
218 setPinsAsOutput(getOutputPins()); // set the output pins as OUTPUT
219 setPinsAsInputPullup(getInputPins()); // set the input pins as INPUT_PULLUP
220
221 for (uint8_t i = 0; i < NO_OF_DUMPLOADS; ++i)
222 {
223 loadPrioritiesAndState[i] = loadPrioritiesAtStartup[i];
224 loadPrioritiesAndState[i] &= loadStateMask;
225 }
226 }
227
228 // First stop the ADC
229 bit_clear(ADCSRA, ADEN);
230
231 // Activate free-running mode
232 ADCSRB = 0x00;
233
234 // Set up the ADC to be free-running
235 bit_set(ADCSRA, ADPS0); // Set the ADC's clock to system clock / 128
236 bit_set(ADCSRA, ADPS1);
237 bit_set(ADCSRA, ADPS2);
238
239 bit_set(ADCSRA, ADATE); // set the Auto Trigger Enable bit in the ADCSRA register. Because
240 // bits ADTS0-2 have not been set (i.e. they are all zero), the
241 // ADC's trigger source is set to "free running mode".
242
243 bit_set(ADCSRA, ADIE); // set the ADC interrupt enable bit. When this bit is written
244 // to one and the I-bit in SREG is set, the
245 // ADC Conversion Complete Interrupt is activated.
246
247 bit_set(ADCSRA, ADEN); // Enable the ADC
248
249 bit_set(ADCSRA, ADSC); // start ADC manually first time
250
251 sei(); // Enable Global Interrupts
252}

391void processGridCurrentRawSample(const int16_t rawSample)
392{
393 // extra items for an LPF to improve the processing of data samples from CT1
394 static int32_t lpf_long{}; // new LPF, for offsetting the behaviour of CTx as a HPF
395
396 // First, deal with the power at the grid connection point (as measured via CT1)
397 // remove most of the DC offset from the current sample (the precise value does not matter)
398 int32_t sampleIminusDC_grid = ((int32_t)(rawSample - DCoffset_I)) << 8;
399 //
400 // extra filtering to offset the HPF effect of CT1
401 const int32_t last_lpf_long = lpf_long;
402 lpf_long += alpha * (sampleIminusDC_grid - last_lpf_long);
403 sampleIminusDC_grid += (lpf_gain * lpf_long);
404
405 // calculate the "real power" in this sample pair and add to the accumulated sum
406 const int32_t filtV_div4 = sampleVminusDC_long >> 2; // reduce to 16-bits (now x64, or 2^6)
407 const int32_t filtI_div4 = sampleIminusDC_grid >> 2; // reduce to 16-bits (now x64, or 2^6)
408 int32_t instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
409 instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
410 sumP_grid += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
411 sumP_grid_overDL_Period += instP;
412}

505void processRawSamples()
506{
507 if (polarityConfirmed == Polarities::POSITIVE)
508 {
509 if (polarityConfirmedOfLastSampleV != Polarities::POSITIVE)
510 {
511 // This is the start of a new +ve half cycle (just after the zero-crossing point)
512 if (beyondStartUpPeriod)
513 {
514 processPlusHalfCycle();
515
516 processStartNewCycle();
517
518 if (EDD_isActive) // Energy Diversion Display
519 {
520 // For diverted energy, the latest contribution needs to be added to an
521 // accumulator which operates with maximum precision.
522
523 if (realEnergy_diverted < antiCreepLimit_inIEUperMainsCycle)
524 {
525 realEnergy_diverted = 0;
526 }
527 divertedEnergyRecent_IEU += realEnergy_diverted;
528
529 // Whole kWh are then recorded separately
530 if (divertedEnergyRecent_IEU > IEU_per_Wh)
531 {
532 divertedEnergyRecent_IEU -= IEU_per_Wh;
533 if (!b_diversionOff && !b_overrideLoadOn[0])
534 {
535 ++divertedEnergyTotal_Wh;
536 }
537 }
538 }
539 }
540 else
541 {
542 processStartUp();
543 }
544 } // end of processing that is specific to the first Vsample in each +ve half cycle
545
546 // still processing samples where the voltage is Polarities::POSITIVE ...
547 // (in this go-faster code, the action from here has moved to the negative half of the cycle)
548
549 } // end of processing that is specific to samples where the voltage is positive
550 else // the polarity of this sample is negative
551 {
552 if (polarityConfirmedOfLastSampleV != Polarities::NEGATIVE)
553 {
554 // This is the start of a new -ve half cycle (just after the zero-crossing point)
555 processMinusHalfCycle();
556 }
557
558 // check to see whether the trigger device can now be reliably armed
559 if (sampleSetsDuringNegativeHalfOfMainsCycle == 3)
560 {
561 if (beyondStartUpPeriod)
562 {
563 /* Determining whether any of the loads need to be changed is is a 3-stage process:
564 * - change the LOGICAL load states as necessary to maintain the energy level
565 * - update the PHYSICAL load states according to the logical -> physical mapping
566 * - update the driver lines for each of the loads.
567 */
568
569 // Restrictions apply for the period immediately after a load has been switched.
570 // Here the recentTransition flag is checked and updated as necessary.
571 if (recentTransition)
572 {
573 if (++postTransitionCount == POST_TRANSITION_MAX_COUNT)
574 {
575 recentTransition = false;
576 }
577 }
578
579 if (energyInBucket_prediction > midPointOfEnergyBucket_long)
580 {
581 // the energy state is in the upper half of the working range
582 lowerEnergyThreshold = lowerThreshold_default; // reset the "opposite" threshold
583 if (energyInBucket_prediction > upperEnergyThreshold)
584 {
585 // Because the energy level is high, some action may be required
586 proceedHighEnergyLevel();
587 }
588 }
589 else
590 { // the energy state is in the lower half of the working range
591 upperEnergyThreshold = upperThreshold_default; // reset the "opposite" threshold
592 if (energyInBucket_prediction < lowerEnergyThreshold)
593 {
594 // Because the energy level is low, some action may be required
595 proceedLowEnergyLevel();
596 }
597 }
598
599 updatePhysicalLoadStates(); // allows the logical-to-physical mapping to be changed
600
601 // update each of the physical loads
602 updatePortsStates();
603
604 // update the Energy Diversion Detector
605 if (loadPrioritiesAndState[0] & loadStateOnBit)
606 {
607 absenceOfDivertedEnergyCount = 0;
608 EDD_isIdle = false;
609 EDD_isActive = true;
610 }
611 else
612 {
613 EDD_isIdle = true;
614 }
615
616 // Now that the energy-related decisions have been taken, min and max limits can now
617 // be applied to the level of the energy bucket. This is to ensure correct operation
618 // when conditions change, i.e. when import changes to export, and vice versa.
619 //
620 if (energyInBucket_long > capacityOfEnergyBucket_long)
621 {
622 energyInBucket_long = capacityOfEnergyBucket_long;
623 }
624 else if (energyInBucket_long < 0)
625 {
626 energyInBucket_long = 0;
627 }
628 }
629 }
630
631 ++sampleSetsDuringNegativeHalfOfMainsCycle;
632 } // end of processing that is specific to samples where the voltage is negative
633 refreshDisplay();
634}

817void processMinusHalfCycle()
818{
819 // This is the start of a new -ve half cycle (just after the zero-crossing point)
820 // which is a convenient point to update the Low Pass Filter for DC-offset removal
821 // The portion which is fed back into the integrator is approximately one percent
822 // of the average offset of all the Vsamples in the previous mains cycle.
823 //
824 long previousOffset = DCoffset_V_long;
825 DCoffset_V_long = previousOffset + (cumVdeltasThisCycle_long >> 12);
826 cumVdeltasThisCycle_long = 0;
827
828 // To ensure that the LPF will always start up correctly when 240V AC is available, its
829 // output value needs to be prevented from drifting beyond the likely range of the
830 // voltage signal. This avoids the need to use a HPF as was done for initial Mk2 builds.
831 //
832 if (DCoffset_V_long < DCoffset_V_min)
833 {
834 DCoffset_V_long = DCoffset_V_min;
835 }
836 else if (DCoffset_V_long > DCoffset_V_max)
837 {
838 DCoffset_V_long = DCoffset_V_max;
839 }
840
841 // The average power that has been measured during the first half of this mains cycle can now be used
842 // to predict the energy state at the end of this mains cycle. That prediction will be used to alter
843 // the state of the load as necessary. The arming signal for the triac can't be set yet - that must
844 // wait until the voltage has advanced further beyond the -ve going zero-crossing point.
845 //
846 long averagePower = sumP_grid / sampleSetsDuringThisMainsCycle; // for 1st half of this mains cycle only
847 //
848 // To avoid repetitive and unnecessary calculations, the increase in energy during each mains cycle is
849 // deemed to be numerically equal to the average power. The predicted value for the energy state at the
850 // end of this mains cycle will therefore be the known energy state at its start plus the average power
851 // as measured. Although the average power has been determined over only half a mains cycle, the correct
852 // number of contributing sample sets has been used so the result can be expected to be a true measurement
853 // of average power, not half of it.
854 //
855 energyInBucket_prediction = energyInBucket_long + averagePower; // at end of this mains cycle
856}

872void processLatestContribution()
873{
874 // Calculate the real power and energy during the last whole mains cycle.
875 //
876 // sumP contains the sum of many individual calculations of instantaneous power. In
877 // order to obtain the average power during the relevant period, sumP must first be
878 // divided by the number of samples that have contributed to its value.
879 //
880 // The next stage would normally be to apply a calibration factor so that real power
881 // can be expressed in Watts. That's fine for floating point maths, but it's not such
882 // a good idea when integer maths is being used. To keep the numbers large, and also
883 // to save time, calibration of power is omitted at this stage. Real Power (stored as
884 // a 'long') is therefore (1/powerCal) times larger than the actual power in Watts.
885 //
886 int32_t realPower_grid = sumP_grid / sampleSetsDuringThisMainsCycle; // proportional to Watts
887 int32_t realPower_diverted = sumP_diverted / sampleSetsDuringThisMainsCycle; // proportional to Watts
888
889 realPower_grid -= requiredExportPerMainsCycle_inIEU; // <- useful for PV simulation
890
891 // Next, the energy content of this power rating needs to be determined. Energy is
892 // power multiplied by time, so the next step would normally be to multiply the measured
893 // value of power by the time over which it was measured.
894 // Instantaneous power is calculated once every mains cycle. When integer maths is
895 // being used, a repetitive power-to-energy conversion seems an unnecessary workload.
896 // As all sampling periods are of similar duration, it is more efficient to just
897 // add all of the power samples together, and note that their sum is actually
898 // SUPPLY_FREQUENCY greater than it would otherwise be.
899 // Although the numerical value itself does not change, I thought that a new name
900 // may be helpful so as to minimise confusion.
901 // The 'energy' variable below is SUPPLY_FREQUENCY * (1/powerCal) times larger than
902 // the actual energy in Joules.
903 //
904 realEnergy_grid = realPower_grid;
905 realEnergy_diverted = realPower_diverted;
906
907 // Energy contributions from the grid connection point (CT1) are summed in an
908 // accumulator which is known as the energy bucket. The purpose of the energy bucket
909 // is to mimic the operation of the supply meter. The range over which energy can
910 // pass to and fro without loss or charge to the user is known as its 'sweet-zone'.
911 // The capacity of the energy bucket is set to this same value within setup().
912 //
913 // The latest contribution can now be added to this energy bucket
914 energyInBucket_long += realEnergy_grid;
915
916 // After a pre-defined period of inactivity, the 4-digit display needs to
917 // close down in readiness for the next's day's data.
918 //
919 if (absenceOfDivertedEnergyCount > displayShutdown_inSeconds)
920 {
921 // clear the accumulators for diverted energy
922 divertedEnergyTotal_Wh = 0;
923 divertedEnergyRecent_IEU = 0;
924 EDD_isActive = false; // energy diversion detector is now inactive
925 }
926 copyOf_divertedEnergyTotal_Wh = divertedEnergyTotal_Wh;
927
928 b_newCycle = true; // a 50 Hz 'tick' for use by the main code
929}

946void proceedHighEnergyLevel()
947{
948 bool bOK_toAddLoad{ true };
949 const auto tempLoad{ nextLogicalLoadToBeAdded() };
950
951 if (tempLoad == NO_OF_DUMPLOADS)
952 {
953 return;
954 }
955
956 // a load which is now OFF has been identified for potentially being switched ON
957 if (recentTransition)
958 {
959 // During the post-transition period, any increase in the energy level is noted.
960 upperEnergyThreshold = energyInBucket_prediction;
961
962 // the energy thresholds must remain within range
963 if (upperEnergyThreshold > capacityOfEnergyBucket_long)
964 {
965 upperEnergyThreshold = capacityOfEnergyBucket_long;
966 }
967
968 // Only the active load may be switched during this period. All other loads must
969 // wait until the recent transition has had sufficient opportunity to take effect.
970 bOK_toAddLoad = (tempLoad == activeLoad);
971 }
972
973 if (bOK_toAddLoad)
974 {
975 loadPrioritiesAndState[tempLoad] |= loadStateOnBit;
976 activeLoad = tempLoad;
977 postTransitionCount = 0;
