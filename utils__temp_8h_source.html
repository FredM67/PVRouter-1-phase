149 int16_t readTemperature(const uint8_t idx) const
150 {
151 static ScratchPad buf;
152
153 if constexpr (TEMP_SENSOR_PRESENT)
154 {
155 if (!oneWire.reset())
156 {
157 return DEVICE_DISCONNECTED_RAW;
158 }
159 oneWire.select(sensorAddrs[idx].addr);
160 oneWire.write(READ_SCRATCHPAD);
161
162 for (auto& buf_elem : buf)
163 {
164 buf_elem = oneWire.read();
165 }
166
167 if (!oneWire.reset())
168 {
169 return DEVICE_DISCONNECTED_RAW;
170 }
171 if (oneWire.crc8(buf, 8) != buf[8])
172 {
173 return DEVICE_DISCONNECTED_RAW;
174 }
175 }
176
177 // result is temperature x16, multiply by 6.25 to convert to temperature x100
178 int16_t result = (buf[1] << 8) | buf[0];
179 result = (result * 6) + (result >> 2);
180 if (result <= TEMP_RANGE_LOW || result >= TEMP_RANGE_HIGH)
181 {
182 return OUTOFRANGE_TEMPERATURE; // return value ('Out of range')
183 }
184
185 return result;
186 }
