98uint8_t *get_tile_from_xbm(uint8_t tx, uint8_t ty, uint8_t xbm_byte_width, const unsigned char *xbm)
99{
100 uint8_t d[8];
101 const auto *p{ xbm + tx + xbm_byte_width * (ty << 3) };
102
103 // Clear the tile buffer
104 memset(d, 0, sizeof(d));
105
106 for (uint8_t i = 0; i < 8; ++i)
107 {
108 uint8_t b = u8x8_pgm_read(p);
109 for (uint8_t bit = 0; bit < 8; ++bit)
110 {
111 if (b & (1 << bit))
112 {
113 d[bit] |= (1 << i);
114 }
115 }
116 p += xbm_byte_width;
117 }
118 return d;
119}

132void u8x8_draw_xbm(uint8_t tx, uint8_t ty, uint8_t xbm_width, uint8_t xbm_height, const unsigned char *xbm)
133{
134 for (uint8_t y = 0; y < (xbm_height >> 3); ++y)
135 {
136 for (uint8_t x = 0; x < (xbm_width >> 3); ++x)
137 {
138 const auto tile{ get_tile_from_xbm(x, y, xbm_width >> 3, xbm) };
139 u8x8.drawTile(tx + x, ty + y, 1, tile);
