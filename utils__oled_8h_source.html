136uint8_t *get_tile_from_xbm(uint8_t tx, uint8_t ty, uint8_t xbm_byte_width, const unsigned char *xbm)
137{
138 uint8_t d[8];
139 const auto *p{ xbm + tx + xbm_byte_width * (ty << 3) };
140
141 // Clear the tile buffer
142 memset(d, 0, sizeof(d));
143
144 for (uint8_t i = 0; i < 8; ++i)
145 {
146 uint8_t b = u8x8_pgm_read(p);
147 for (uint8_t bit = 0; bit < 8; ++bit)
148 {
149 if (b & (1 << bit))
150 {
151 d[bit] |= (1 << i);
152 }
153 }
154 p += xbm_byte_width;
155 }
156 return d;
157}
