263
264 serializeJson(doc, Serial);
265 Serial.println();

291void sendTelemetryData()
292{
293 static TeleInfo teleInfo;
294
295 teleInfo.startFrame(); // Start a new telemetry frame
296
297 teleInfo.send("P", tx_data.powerGrid); // Send power grid data
298
299 if constexpr (RELAY_DIVERSION)
300 {
301 teleInfo.send("R", static_cast< int16_t >(relays.get_average())); // Send relay average if diversion is enabled
302
303 uint8_t idx = 0;
304 do
305 {
306 teleInfo.send("R", relays.get_relay(idx).isRelayON()); // Send diverted energy count for each relay
307 } while (++idx < relays.get_size());
308 }
309
310 teleInfo.send("D", tx_data.powerDiverted); // Send power diverted
311 teleInfo.send("E", static_cast< int16_t >(divertedEnergyTotal_Wh)); // Send diverted energy in Wh
312 teleInfo.send("V", tx_data.Vrms_L_x100); // Send voltage in volts
313
314 if constexpr (TEMP_SENSOR_PRESENT)
315 {
316 for (uint8_t idx = 0; idx < temperatureSensing.get_size(); ++idx)
317 {
318 if ((OUTOFRANGE_TEMPERATURE == tx_data.temperature_x100[idx])
319 || (DEVICE_DISCONNECTED_RAW == tx_data.temperature_x100[idx]))
320 {
321 continue; // Skip invalid temperature readings
322 }
323 teleInfo.send("T", tx_data.temperature_x100[idx], idx + 1); // Send temperature
324 }
325 }
326
327 teleInfo.send("N", static_cast< int16_t >(absenceOfDivertedEnergyCount)); // Send absence of diverted energy count for 50Hz
328
329 teleInfo.send("S", copyOf_sampleSetsDuringThisDatalogPeriod);
330 teleInfo.send("S_MC", copyOf_lowestNoOfSampleSetsPerMainsCycle);
331
332 teleInfo.endFrame(); // Finalize and send the telemetry frame
333}

352inline void sendResults(bool bOffPeak)
353{

355

357 {
358 startup = false;
359 return; // reject the first datalogging which is incomplete !
360 }

364#endif
365
366 if constexpr (SERIAL_OUTPUT_TYPE == SerialOutputType::HumanReadable)
367 {
368 printForSerialText();
369 }
370 else if constexpr (SERIAL_OUTPUT_TYPE == SerialOutputType::IoT)
371 {
372 sendTelemetryData();
373 }
374 else if constexpr (SERIAL_OUTPUT_TYPE == SerialOutputType::EmonCMS)
375 {
376 printForEmonCMS(bOffPeak);
377 }
378}
